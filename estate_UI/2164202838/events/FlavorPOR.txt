	fire_only_once = yes
	
	trigger = {
		tag = POR
		exploration_ideas = 2

		is_year = 1470
		OR = {
			NOT = {
				north_america = {
					has_discovered = ROOT
				}
			}
			NOT = {
				south_america = {
					has_discovered = ROOT
				}
			}
		}
	}

		months = 1
	}
	
	
	option = {		# Reject his proposal
		name = "flavor_por.EVTOPTA1"

			limit = {
				exists = SPA

			}
			SPA = {
				country_event = { id = flavor_spa.1 days = 10 }
			}
		}
		if = {
			limit = {
				NOT = { exists = SPA }
				exists = CAS
				CAS = { has_idea = quest_for_the_new_world }
			}
			CAS = {
				country_event = { id = flavor_spa.1 days = 10 }
			}
		}
		if = {
			limit = {
				OR = {
					AND = {
						NOT = { exists = SPA }
						NOT = { exists = CAS }
					}
					AND = {
						CAS = { NOT = { has_idea = quest_for_the_new_world } }
						SPA = { NOT = { has_idea = quest_for_the_new_world } }
					}
				}
				any_known_country = {
					has_idea = quest_for_the_new_world
